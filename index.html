// ========================================== // SMART CSV INGESTION LOGIC (UPDATED) // ========================================== let uploadedData = { customers: null, vehicles: null }; function handleCSVUpload(event, type) { const file = event.target.files[0]; if (!file) return; Papa.parse(file, { header: true, dynamicTyping: true, skipEmptyLines: true, complete: function(results) { if (type === 'customers') { uploadedData.customers = results.data; el("btnCustFile").textContent = "✅ Leads Loaded"; } else { uploadedData.vehicles = results.data; el("btnVehFile").textContent = "✅ Vehicles Loaded"; } // If we have vehicles, we can run (since the vehicle file has active customers in it) if (uploadedData.vehicles) { el("btnRunDashboard").disabled = false; el("btnRunDashboard").classList.add("good"); } } }); } // Helper to convert DD/MM/YYYY to YYYY-MM-DD for the dashboard charts function parseDDMMYYYY(dateStr) { if (!dateStr) return null; const str = dateStr.toString().trim(); const parts = str.split('/'); if (parts.length === 3) { return `${parts[2]}-${parts[1].padStart(2, '0')}-${parts[0].padStart(2, '0')}`; } return str; // Fallback if already ISO or different } function ingestRealData() { const baseRateAnnual = 8200; // Fallback rate if none provided let realVehicles = []; let realCustomers = []; let customerIdCounter = 1; // 1. Map the Vehicle CSV (Flat File) uploadedData.vehicles.forEach((row, index) => { // We use a helper to find keys that "contain" the words, avoiding exact match errors const keys = Object.keys(row); const getVal = (match) => { const key = keys.find(k => k.toLowerCase().includes(match.toLowerCase())); return key ? row[key] : null; }; const vin = getVal('VIN'); if (!vin) return; // Skip empty rows // Determine Deal Status const rawStatus = (getVal('Deal Status') || '').toString().toUpperCase(); const isActive = rawStatus === 'ACTIVE'; // Build the Vehicle Object const v = { id: "V" + pad(index + 1, 4), vin: vin, type_make_model: `${getVal('Brand') || ''} ${getVal('Model') || ''} ${getVal('Model Year') || ''}`.trim(), plate: getVal('Plate Number') || '—', diamond: getVal('Diamond Num') || '—', diamond_expiration_date: parseDDMMYYYY(getVal('Diamond Exp')), has_active_deal: isActive, deal_number: getVal('Current Lease') || null, deal_start_date: parseDDMMYYYY(getVal('Start Deal')), deal_end_date: parseDDMMYYYY(getVal('End Deal')), sales_channel: getVal('Sales Channel') || null, vehicle_location: getVal('Physical Loc') || getVal('Current Loc') || 'Unknown', different_drivers_ytd: isActive ? 1 : 0, drivers_switched_ytd: 0, insurance_annual_rate: baseRateAnnual, // Hardcoded fallback for now insurance_daily_rate: Math.round((baseRateAnnual / 365) * 100) / 100, insurance_rate_source: isActive ? "driver" : "base", current_driver_customer_id: null, last_driver_customer_id: null }; // Extract the Driver from this row (if present) const custName = getVal('Customer Name'); if (custName && custName.trim() !== '') { const c = { id: "C" + pad(customerIdCounter++, 4), name: custName, dl_number: getVal('Customer DL') || '—', tlc_number: getVal('Customer TLC') || '—', insurance_annual_rate: baseRateAnnual, insurance_daily_rate: Math.round((baseRateAnnual / 365) * 100) / 100, rate_tier: rateTier(baseRateAnnual), has_active_deal: isActive, deal_number: v.deal_number, vin: v.vin, is_lead: false, lead_stage: null }; realCustomers.push(c); v.current_driver_customer_id = c.id; } realVehicles.push(v); }); // 2. If a separate Customers (Leads) file was uploaded, add them if (uploadedData.customers) { uploadedData.customers.forEach(row => { // Map your separate leads file fields here if needed const c = { id: "L" + pad(customerIdCounter++, 4), name: row['Customer Name'] || row['Name'] || 'Unknown Lead', dl_number: row['Customer DL'] || row['DL #'] || '—', tlc_number: row['Customer TLC'] || '—', insurance_annual_rate: Number(row['Insurance Rate']) || 7000, insurance_daily_rate: Math.round(((Number(row['Insurance Rate']) || 7000) / 365) * 100) / 100, rate_tier: rateTier(Number(row['Insurance Rate']) || 7000), has_active_deal: false, vin: null, is_lead: true, lead_stage: "Ready to match" }; realCustomers.push(c); }); } // 3. Index and generate historical fallback data for charts const customersById = Object.fromEntries(realCustomers.map(c => [c.id, c])); const vehiclesByVIN = Object.fromEntries(realVehicles.map(v => [v.vin, v])); const rng = Math.random; realVehicles.forEach(v => { v.rate_history_120d = buildRateHistory(rng, v.insurance_rate_source, baseRateAnnual, v.insurance_annual_rate, v.drivers_switched_ytd); const {mapByDate, dlChangesCount} = buildDriverTimeline(rng, v, baseRateAnnual, customersById); v.daily_map_120d = mapByDate; v.dl_changes_120d = dlChangesCount; }); // Overwrite DB State DB = { meta: { seed: "REAL_CSV_DATA", generated_at: new Date().toISOString(), customers_count: realCustomers.length, vehicles_count: realVehicles.length, base_rate_annual: baseRateAnnual }, customers: realCustomers, vehicles: realVehicles, index: { customersById, vehiclesByVIN } }; OPPS = buildOpportunities(); refreshAll(); toast("✅ Real Data Loaded and Dashboard Updated!"); } // Hook up event listeners document.getElementById('uploadCustomers').addEventListener('change', (e) => handleCSVUpload(e, 'customers')); document.getElementById('uploadVehicles').addEventListener('change', (e) => handleCSVUpload(e, 'vehicles')); document.getElementById('btnRunDashboard').addEventListener('click', ingestRealData);
