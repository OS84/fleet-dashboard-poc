function fmtRateAnnual(n){return Math.round(n/10)*10}

function buildRateHistory(rng, mode, baseAnnual, targetAnnual, switchesYTD){

  const days=120, out=[];

  const end=today0(), start=addDays(end, -(days-1));

  const maxShown=clamp(switchesYTD,0,5);

  const shown=(mode==="base")?0:randInt(rng,0,maxShown);

  const switchOffsets=[];

  for(let i=0;i<shown;i++) switchOffsets.push(randInt(rng,8,days-8));

  switchOffsets.sort((a,b)=>a-b);

  let segAnnual=(mode==="base")?baseAnnual:fmtRateAnnual(randInt(rng,6000,10000));

  let next=0;

  for(let i=0;i<days;i++){

    let switched=false;

    if(next<switchOffsets.length && i===switchOffsets[next]){

      segAnnual=fmtRateAnnual(randInt(rng,6000,10000));

      switched=true;

      next++;

    }

    if(i>=days-10) segAnnual=targetAnnual;

    const d=addDays(start,i);

    const daily=Math.round((segAnnual/365)*100)/100;

    out.push({date:dateISO(d), annual:segAnnual, daily, switched});

  }

  return out;

}

function buildDriverTimeline(rng, vehicle, baseAnnual, customersById){

  const history = vehicle.rate_history_120d || [];

  const map = {};

  if(history.length===0) return {mapByDate:map, dlChangesCount:0};


  let dlChanges=0;

  let currentDriverId = vehicle.current_driver_customer_id || null;

  let lastDriverId = vehicle.last_driver_customer_id || null;


  const changePoints = [];

  if(vehicle.has_active_deal && currentDriverId && vehicle.drivers_switched_ytd>0){

    const changes = clamp(vehicle.drivers_switched_ytd,1,4);

    for(let i=0;i<changes;i++){ changePoints.push(randInt(rng,10,history.length-10)); }

    changePoints.sort((a,b)=>a-b);

  }

  const altDrivers = [];

  for(let i=0;i<changePoints.length;i++){

    const alt = customersById["C"+pad(randInt(rng,1,1075),4)];

    altDrivers.push(alt ? alt.id : currentDriverId);

  }


  let segIdx=-1;

  let segDriverId=null;


  for(let i=0;i<history.length;i++){

    if(segIdx+1 < changePoints.length && i===changePoints[segIdx+1]){

      segIdx++; segDriverId=altDrivers[segIdx]; dlChanges++;

    }


    let driverId=null, dl=null, annual=vehicle.insurance_annual_rate, source=vehicle.insurance_rate_source;


    if(vehicle.has_active_deal){

      driverId = segDriverId || currentDriverId;

      const c = driverId ? customersById[driverId] : null;

      dl = c ? c.dl_number : null;

      annual = history[i].annual;

      source="driver";

    } else if(lastDriverId){

      const c = customersById[lastDriverId];

      dl = c ? c.dl_number : null;

      annual = history[i].annual;

      source="last_driver";

    } else {

      dl = null;

      annual = baseAnnual;

      source="base";

    }


    map[history[i].date] = {date: history[i].date, daily: history[i].daily, annual, dl, driverId, source};

  }


  return {mapByDate: map, dlChangesCount: dlChanges};

}

function computeSpend(){

  const vehicles=DB.vehicles;

  return {dailySpendAll, spentYTD: dailySpendAll*daysYTD, spentMTD: dailySpendAll*daysMTD};

}

function computeMoneyOnFloor(){

  const vehicles=DB.vehicles;

  const vacant=vehicles.filter(v=>!v.has_active_deal && !v.last_driver_customer_id);

  const vacantDaily=vacant.reduce((s,v)=>s+v.insurance_daily_rate,0);

}

      kind:item.kind, vin:v.vin, vehicle:v.type_make_model, location:v.vehicle_location, deal_end:v.deal_end_date,

      lead_daily: leadDaily, savings_daily: savingsDaily, savings_30d: savingsDaily!=null ? savingsDaily*30 : null

    });

  }

  out.sort((a,b)=> (b.savings_daily||-1) - (a.savings_daily||-1)); return out;

}

function drawLineChart(canvas, series){

  const ctx=canvas.getContext("2d"), w=canvas.width, h=canvas.height; ctx.clearRect(0,0,w,h);

  if(!series || series.length===0) return null;

  const ys=series.map(p=>p.y), minV=Math.min(...ys), maxV=Math.max(...ys), padV=(maxV-minV)===0?1:(maxV-minV)*0.12, lo=minV-padV, hi=maxV+padV;

  ctx.globalAlpha=0.85; ctx.strokeStyle="rgba(231,236,255,.18)"; ctx.lineWidth=1;

  for(let g=1;g<=3;g++){ const y=(h-20)*(g/4)+8; ctx.beginPath(); ctx.moveTo(8,y); ctx.lineTo(w-8,y); ctx.stroke(); }

  ctx.globalAlpha=1; ctx.strokeStyle="rgba(106,228,255,.95)"; ctx.lineWidth=2; ctx.beginPath();

  for(let i=0;i<series.length;i++){ const x=8+(i/(series.length-1))*(w-16), y=8+(1-((series[i].y-lo)/(hi-lo)))*(h-28); if(i===0) ctx.moveTo(x,y); else ctx.lineTo(x,y); } ctx.stroke();

  ctx.strokeStyle="rgba(255,209,102,.65)"; ctx.lineWidth=1; let dlChanges=0;

  for(let i=0;i<series.length;i++){ if(series[i].dlChanged){ dlChanges++; const x=8+(i/(series.length-1))*(w-16); ctx.beginPath(); ctx.moveTo(x, 10); ctx.lineTo(x, h-22); ctx.stroke(); } }

  const lastIdx=series.length-1, lx=8+(lastIdx/(series.length-1))*(w-16), ly=8+(1-((series[lastIdx].y-lo)/(hi-lo)))*(h-28); ctx.fillStyle="rgba(155,140,255,.95)"; ctx.beginPath(); ctx.arc(lx,ly,3.8,0,Math.PI*2); ctx.fill();

  return {minV, maxV, lastV: series[lastIdx].y, dlChanges};

}
